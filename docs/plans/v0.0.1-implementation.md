> **For agentic workers:** REQUIRED: Follow the `test-driven-development` skill for every task. Use `coding:execute` to implement task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a K8s controller that watches Pending pods on ACK and emits per-nodepool diagnostic Events visible via kubectl describe pod.

**Architecture:** Go controller using controller-runtime. Watches Pending pods, groups nodes by ACK nodepool label, checks scheduling fit per nodepool, reads cluster-autoscaler-status ConfigMap for scaling state, emits NodepoolAccepted/NodepoolRejected/NodepoolNoStock/NodepoolCandidate Events on the pod.

**Tech Stack:** Go 1.22+, controller-runtime v0.19.x, envtest for testing

```go
import (

	clientgoscheme "k8s.io/client-go/kubernetes/scheme"

	"github.com/helmi/fitcheck/internal/controller"

var scheme = runtime.NewScheme()

func init() {
	utilruntime.Must(clientgoscheme.AddToScheme(scheme))
}

func main() {
	var (
		nodepoolLabel       string
		recheckInterval     string
		initialDelay        string
		namespace           string
		autoscalerConfigMap string
		metricsAddr         string
	)

	flag.StringVar(&nodepoolLabel, "nodepool-label", "alibabacloud.com/nodepool-id", "Label key for grouping nodes into nodepools")
	flag.StringVar(&recheckInterval, "recheck-interval", "30s", "Re-evaluation interval for pending pods")
	flag.StringVar(&initialDelay, "initial-delay", "10s", "Delay before first diagnosis")
	flag.StringVar(&namespace, "namespace", "", "Restrict to specific namespace (empty = all)")
	flag.StringVar(&autoscalerConfigMap, "autoscaler-configmap", "cluster-autoscaler-status", "ConfigMap name for autoscaler status")
	flag.StringVar(&metricsAddr, "metrics-addr", ":8080", "Metrics bind address")

	opts := zap.Options{Development: true}
	opts.BindFlags(flag.CommandLine)
	flag.Parse()

	ctrl.SetLogger(zap.New(zap.UseFlagOptions(&opts)))
	log := ctrl.Log.WithName("setup")

	recheckDuration, err := parseTimeDuration(recheckInterval)
	if err != nil {
		log.Error(err, "invalid --recheck-interval")
		os.Exit(1)
	}
	initialDelayDuration, err := parseTimeDuration(initialDelay)
	if err != nil {
		log.Error(err, "invalid --initial-delay")
		os.Exit(1)
	}

	cacheOpts := cache.Options{
		ByObject: map[client.Object]cache.ByObject{
			&corev1.ConfigMap{}: {
				Namespaces: map[string]cache.Config{
					"kube-system": {},
				},
			},
		},
	}

	mgr, err := ctrl.NewManager(ctrl.GetConfigOrDie(), ctrl.Options{
		Scheme:  scheme,
		Cache:   cacheOpts,
		Metrics: metricsserver.Options{BindAddress: metricsAddr},
	})
	if err != nil {
		log.Error(err, "unable to create manager")
		os.Exit(1)
	}

	if err := controller.SetupPodReconciler(mgr, controller.PodReconcilerOpts{
		NodepoolLabel:       nodepoolLabel,
		RecheckInterval:     recheckDuration,
		InitialDelay:        initialDelayDuration,
		AutoscalerConfigMap: autoscalerConfigMap,
	}); err != nil {
		log.Error(err, "unable to setup controller")
		os.Exit(1)
	}

	log.Info("starting manager")
	if err := mgr.Start(ctrl.SetupSignalHandler()); err != nil {
		log.Error(err, "manager exited with error")
		os.Exit(1)
	}
}
```

```go
package types

import (
	"fmt"

	corev1 "k8s.io/api/core/v1"
	"k8s.io/apimachinery/pkg/api/resource"
)

// Verdict represents the scheduling verdict for a nodepool.
type Verdict string

const (
	VerdictAccepted  Verdict = "Accepted"
	VerdictCandidate Verdict = "Candidate"
	VerdictRejected  Verdict = "Rejected"
	VerdictNoStock   Verdict = "NoStock"
)

// RejectionCategory defines the order in which scheduling dimensions are checked.
type RejectionCategory int

const (
	CategoryTaint        RejectionCategory = 1
	CategoryNodeSelector RejectionCategory = 2
	CategoryAffinity     RejectionCategory = 3
	CategoryResources    RejectionCategory = 4
)

// NodepoolDiagnosis is the result of diagnosing scheduling fit for one nodepool.
type NodepoolDiagnosis struct {
	NodepoolName      string
	Verdict           Verdict
	RejectionCategory RejectionCategory
	Reason            string
	AutoscalerStatus  *AutoscalerNodeGroupStatus
	FittingNodes      int
	TotalNodes        int
}

// Message formats the diagnosis as the event message body.
func (d NodepoolDiagnosis) Message() string {
	return fmt.Sprintf("nodepool/%s: %s", d.NodepoolName, d.Reason)
}

// EventType returns the Kubernetes event type (Normal or Warning).
func (d NodepoolDiagnosis) EventType() string {
	switch d.Verdict {
	case VerdictAccepted, VerdictCandidate:
		return string(corev1.EventTypeNormal)
	default:
		return string(corev1.EventTypeWarning)
	}
}

// EventReason returns the Kubernetes event reason string.
func (d NodepoolDiagnosis) EventReason() string {
	switch d.Verdict {
	case VerdictAccepted:
		return "NodepoolAccepted"
	case VerdictCandidate:
		return "NodepoolCandidate"
	case VerdictRejected:
		return "NodepoolRejected"
	case VerdictNoStock:
		return "NodepoolNoStock"
	default:
		return "NodepoolRejected"
	}
}

// ScaleUpStatus represents the autoscaler scale-up status for a nodegroup.
type ScaleUpStatus string

const (
	ScaleUpNeeded     ScaleUpStatus = "Needed"
	ScaleUpNotNeeded  ScaleUpStatus = "NotNeeded"
	ScaleUpInProgress ScaleUpStatus = "InProgress"
	ScaleUpNoActivity ScaleUpStatus = "NoActivity"
	ScaleUpBackoff    ScaleUpStatus = "Backoff"
)

// ScaleDownStatus represents the autoscaler scale-down status.
type ScaleDownStatus string

const (
	ScaleDownNoCandidates     ScaleDownStatus = "NoCandidates"
	ScaleDownCandidatesPresent ScaleDownStatus = "CandidatesPresent"
)

// HealthStatus represents the health status of a nodegroup.
type HealthStatus string

const (
	HealthHealthy   HealthStatus = "Healthy"
	HealthUnhealthy HealthStatus = "Unhealthy"
)

// AutoscalerStatus is the parsed content of the cluster-autoscaler-status ConfigMap.
type AutoscalerStatus struct {
	Time          string
	Status        string
	ClusterHealth HealthStatus
	ClusterScaleUp  ScaleUpStatus
	ClusterScaleDown ScaleDownStatus
	NodeGroups    []AutoscalerNodeGroupStatus
}

// AutoscalerNodeGroupStatus holds the autoscaler state for one nodegroup (ESS scaling group).
type AutoscalerNodeGroupStatus struct {
	Name               string // ESS Scaling Group ID (asg-xxx)
	Health             HealthStatus
	ReadyNodes         int
	CloudProviderTarget int
	MinSize            int
	MaxSize            int
	ScaleUp            ScaleUpStatus
	ScaleDown          ScaleDownStatus
	BackoffErrorCode   string
	BackoffErrorMessage string
}

// IsAtMaxSize returns true if the nodegroup is already at max capacity.
func (s AutoscalerNodeGroupStatus) IsAtMaxSize() bool {
	return s.CloudProviderTarget >= s.MaxSize
}

// IsBackoff returns true if scale-up is in backoff state.
func (s AutoscalerNodeGroupStatus) IsBackoff() bool {
	return s.ScaleUp == ScaleUpBackoff
}

// IsNoStock returns true if the backoff error indicates cloud stock exhaustion.
func (s AutoscalerNodeGroupStatus) IsNoStock() bool {
	return s.IsBackoff() && containsNoStockSignal(s.BackoffErrorMessage)
}

func containsNoStockSignal(msg string) bool {
	noStockPatterns := []string{
		"OperationDenied.NoStock",
		"out of stock",
	}
	for _, pattern := range noStockPatterns {
		if containsIgnoreCase(msg, pattern) {
			return true
		}
	}
	return false
}

func containsIgnoreCase(s, substr string) bool {
	return len(s) >= len(substr) && searchIgnoreCase(s, substr)
}

func searchIgnoreCase(s, substr string) bool {
	if len(substr) == 0 {
		return true
	}
	for i := 0; i <= len(s)-len(substr); i++ {
		if matchIgnoreCase(s[i:i+len(substr)], substr) {
			return true
		}
	}
	return false
}

func matchIgnoreCase(a, b string) bool {
	for i := 0; i < len(a); i++ {
		ca, cb := a[i], b[i]
		if ca >= 'A' && ca <= 'Z' {
			ca += 32
		}
		if cb >= 'A' && cb <= 'Z' {
			cb += 32
		}
		if ca != cb {
			return false
		}
	}
	return true
}

// NodepoolInfo holds summarized information about a nodepool for diagnosis.
type NodepoolInfo struct {
	Name  string
	Nodes []NodeInfo
}

// NodeInfo holds the scheduling-relevant information about a single node.
type NodeInfo struct {
	Name            string
	NodepoolID      string
	Taints          []corev1.Taint
	Labels          map[string]string
	AllocatableCPU  resource.Quantity
	AllocatableMem  resource.Quantity
	AllocatableGPU  resource.Quantity
	RequestedCPU    resource.Quantity
	RequestedMem    resource.Quantity
	RequestedGPU    resource.Quantity
}

// AvailableCPU returns allocatable minus requested CPU.
func (n NodeInfo) AvailableCPU() resource.Quantity {
	result := n.AllocatableCPU.DeepCopy()
	result.Sub(n.RequestedCPU)
	return result
}

// AvailableMem returns allocatable minus requested memory.
func (n NodeInfo) AvailableMem() resource.Quantity {
	result := n.AllocatableMem.DeepCopy()
	result.Sub(n.RequestedMem)
	return result
}

// AvailableGPU returns allocatable minus requested GPU.
func (n NodeInfo) AvailableGPU() resource.Quantity {
	result := n.AllocatableGPU.DeepCopy()
	result.Sub(n.RequestedGPU)
	return result
}
```

```go
package diagnosis

import (
	"fmt"

	corev1 "k8s.io/api/core/v1"

	"github.com/helmi/fitcheck/internal/types"
)

// checkTaints checks if the pod tolerates all scheduling-relevant taints on the node.
// Returns (reason, rejected). If rejected is false, reason is empty.
func checkTaints(pod *corev1.Pod, node types.NodeInfo) (string, bool) {
	for _, taint := range node.Taints {
		// PreferNoSchedule is a soft constraint, skip it
		if taint.Effect == corev1.TaintEffectPreferNoSchedule {
			continue
		}
		if !toleratesTaint(pod.Spec.Tolerations, taint) {
			return fmt.Sprintf("taint {%s=%s:%s} not tolerated", taint.Key, taint.Value, taint.Effect), true
		}
	}
	return "", false
}

// toleratesTaint checks if any toleration in the list matches the given taint.
func toleratesTaint(tolerations []corev1.Toleration, taint corev1.Taint) bool {
	for _, toleration := range tolerations {
		if tolerationMatchesTaint(toleration, taint) {
			return true
		}
	}
	return false
}

// tolerationMatchesTaint returns true if the toleration matches the taint.
// Implements the same logic as k8s.io/component-helpers/scheduling/corev1.
func tolerationMatchesTaint(toleration corev1.Toleration, taint corev1.Taint) bool {
	// Empty key with Exists operator matches all taints
	if toleration.Key == "" && toleration.Operator == corev1.TolerationOpExists {
		// If effect is empty, it matches all effects
		if toleration.Effect == "" {
			return true
		}
		return toleration.Effect == taint.Effect
	}

	// Key must match
	if toleration.Key != taint.Key {
		return false
	}

	// Effect must match (empty toleration effect matches all)
	if toleration.Effect != "" && toleration.Effect != taint.Effect {
		return false
	}

	// Check operator
	switch toleration.Operator {
	case corev1.TolerationOpExists:
		return true
	case "", corev1.TolerationOpEqual:
		return toleration.Value == taint.Value
	default:
		return false
	}
}
```

```go
// checkNodeAffinity checks if the node satisfies the pod's required node affinity.
// Only checks RequiredDuringSchedulingIgnoredDuringExecution (hard constraint).
// PreferredDuringSchedulingIgnoredDuringExecution is a soft constraint and is ignored.
func checkNodeAffinity(pod *corev1.Pod, node types.NodeInfo) (string, bool) {
	if pod.Spec.Affinity == nil || pod.Spec.Affinity.NodeAffinity == nil {
		return "", false
	}

	required := pod.Spec.Affinity.NodeAffinity.RequiredDuringSchedulingIgnoredDuringExecution
	if required == nil {
		return "", false
	}

	// NodeSelectorTerms are OR'd: at least one term must match
	for _, term := range required.NodeSelectorTerms {
		if matchesNodeSelectorTerm(term, node.Labels) {
			return "", false
		}
	}

	return "node affinity not matched", true
}

// matchesNodeSelectorTerm returns true if the node labels satisfy all expressions in the term.
// Expressions within a term are AND'd.
func matchesNodeSelectorTerm(term corev1.NodeSelectorTerm, labels map[string]string) bool {
	for _, expr := range term.MatchExpressions {
		if !matchesExpression(expr, labels) {
			return false
		}
	}
	for _, expr := range term.MatchFields {
		// MatchFields matches against node fields, not labels.
		// For simplicity in v0.0.1, we skip field matching.
		_ = expr
	}
	return true
}

// matchesExpression evaluates a single NodeSelectorRequirement against labels.
func matchesExpression(expr corev1.NodeSelectorRequirement, labels map[string]string) bool {
	value, exists := labels[expr.Key]

	switch expr.Operator {
	case corev1.NodeSelectorOpIn:
		if !exists {
			return false
		}
		for _, v := range expr.Values {
			if v == value {
				return true
			}
		}
		return false

	case corev1.NodeSelectorOpNotIn:
		if !exists {
			return true
		}
		for _, v := range expr.Values {
			if v == value {
				return false
			}
		}
		return true

	case corev1.NodeSelectorOpExists:
		return exists

	case corev1.NodeSelectorOpDoesNotExist:
		return !exists

	case corev1.NodeSelectorOpGt, corev1.NodeSelectorOpLt:
		// Gt/Lt are for integer comparisons on node fields.
		// Rarely used for scheduling decisions. Skip in v0.0.1.
		return true

	default:
		return false
	}
}
```

```go
import (
	"k8s.io/apimachinery/pkg/api/resource"
)

const gpuResourceName corev1.ResourceName = "nvidia.com/gpu"

// checkResources checks if the node has sufficient available resources for the pod.
func checkResources(pod *corev1.Pod, node types.NodeInfo) (string, bool) {
	podCPU, podMem, podGPU := effectivePodRequests(pod)

	availCPU := node.AvailableCPU()
	availMem := node.AvailableMem()
	availGPU := node.AvailableGPU()

	if podCPU.Cmp(resource.MustParse("0")) > 0 && podCPU.Cmp(availCPU) > 0 {
		return fmt.Sprintf("Insufficient cpu (requested=%s, available=%s)", podCPU.String(), availCPU.String()), true
	}
	if podMem.Cmp(resource.MustParse("0")) > 0 && podMem.Cmp(availMem) > 0 {
		return fmt.Sprintf("Insufficient memory (requested=%s, available=%s)", podMem.String(), availMem.String()), true
	}
	if podGPU.Cmp(resource.MustParse("0")) > 0 && podGPU.Cmp(availGPU) > 0 {
		return fmt.Sprintf("Insufficient gpu (requested=%s, available=%s)", podGPU.String(), availGPU.String()), true
	}

	return "", false
}

// effectivePodRequests computes the effective resource request for a pod.
// For regular containers: sum of all container requests.
// For init containers: max of any single init container.
// Effective = max(sum of regular, max of init) per resource dimension.
func effectivePodRequests(pod *corev1.Pod) (cpu, mem, gpu resource.Quantity) {
	var regularCPU, regularMem, regularGPU resource.Quantity
	var initMaxCPU, initMaxMem, initMaxGPU resource.Quantity

	for _, c := range pod.Spec.Containers {
		if req, ok := c.Resources.Requests[corev1.ResourceCPU]; ok {
			regularCPU.Add(req)
		}
		if req, ok := c.Resources.Requests[corev1.ResourceMemory]; ok {
			regularMem.Add(req)
		}
		if req, ok := c.Resources.Requests[gpuResourceName]; ok {
			regularGPU.Add(req)
		}
	}

	for _, c := range pod.Spec.InitContainers {
		if req, ok := c.Resources.Requests[corev1.ResourceCPU]; ok {
			if req.Cmp(initMaxCPU) > 0 {
				initMaxCPU = req.DeepCopy()
			}
		}
		if req, ok := c.Resources.Requests[corev1.ResourceMemory]; ok {
			if req.Cmp(initMaxMem) > 0 {
				initMaxMem = req.DeepCopy()
			}
		}
		if req, ok := c.Resources.Requests[gpuResourceName]; ok {
			if req.Cmp(initMaxGPU) > 0 {
				initMaxGPU = req.DeepCopy()
			}
		}
	}

	cpu = maxQuantity(regularCPU, initMaxCPU)
	mem = maxQuantity(regularMem, initMaxMem)
	gpu = maxQuantity(regularGPU, initMaxGPU)
	return
}

func maxQuantity(a, b resource.Quantity) resource.Quantity {
	if a.Cmp(b) >= 0 {
		return a
	}
	return b
}
```

```go
import "sort"

// GroupNodesByNodepool groups nodes by their nodepool label value.
// Nodes without the label are grouped under "<unlabeled>".
func GroupNodesByNodepool(nodes []types.NodeInfo, nodepoolLabel string) []types.NodepoolInfo {
	groups := map[string][]types.NodeInfo{}

	for _, node := range nodes {
		poolName := node.NodepoolID
		if poolName == "" {
			poolName = "<unlabeled>"
		}
		groups[poolName] = append(groups[poolName], node)
	}

	pools := make([]types.NodepoolInfo, 0, len(groups))
	for name, nodeList := range groups {
		pools = append(pools, types.NodepoolInfo{
			Name:  name,
			Nodes: nodeList,
		})
	}

	// Sort for deterministic output
	sort.Slice(pools, func(i, j int) bool {
		return pools[i].Name < pools[j].Name
	})

	return pools
}
```

```go
// DiagnoseNodepools runs scheduling fit checks on each nodepool and returns per-nodepool verdicts.
// autoscalerStatus maps nodepool name -> autoscaler status (nil if unavailable).
func DiagnoseNodepools(pod *corev1.Pod, pools []types.NodepoolInfo, autoscalerStatus map[string]*types.AutoscalerNodeGroupStatus) []types.NodepoolDiagnosis {
	results := make([]types.NodepoolDiagnosis, 0, len(pools))

	for _, pool := range pools {
		diag := diagnosePool(pod, pool, autoscalerStatus)
		results = append(results, diag)
	}

	return results
}

func diagnosePool(pod *corev1.Pod, pool types.NodepoolInfo, autoscalerStatus map[string]*types.AutoscalerNodeGroupStatus) types.NodepoolDiagnosis {
	diag := types.NodepoolDiagnosis{
		NodepoolName: pool.Name,
		TotalNodes:   len(pool.Nodes),
	}

	if len(pool.Nodes) == 0 {
		diag.Verdict = types.VerdictRejected
		diag.Reason = "no nodes in pool"
		return diag
	}

	// Check each node against all scheduling dimensions
	type nodeResult struct {
		rejected          bool
		rejectionCategory types.RejectionCategory
		reason            string
	}

	nodeResults := make([]nodeResult, len(pool.Nodes))
	fittingCount := 0

	for i, node := range pool.Nodes {
		nr := checkNodeFit(pod, node)
		nodeResults[i] = nodeResult{
			rejected:          nr.rejected,
			rejectionCategory: nr.category,
			reason:            nr.reason,
		}
		if !nr.rejected {
			fittingCount++
		}
	}

	diag.FittingNodes = fittingCount

	// If any node fits, the pool accepts the pod
	if fittingCount > 0 {
		diag.Verdict = types.VerdictAccepted
		availCPU := resource.Quantity{}
		availMem := resource.Quantity{}
		for _, node := range pool.Nodes {
			nodeFit := checkNodeFit(pod, node)
			if !nodeFit.rejected {
				cpu := node.AvailableCPU()
				mem := node.AvailableMem()
				availCPU.Add(cpu)
				availMem.Add(mem)
			}
		}
		diag.Reason = fmt.Sprintf("%d of %d nodes fit (cpu=%s avail, mem=%s avail)",
			fittingCount, len(pool.Nodes),
			formatQuantity(availCPU), formatQuantity(availMem))
		return diag
	}

	// All nodes rejected. Find the dominant rejection category (most common).
	dominantCategory, dominantReason := dominantRejection(nodeResults)
	diag.RejectionCategory = dominantCategory

	// Check if a fresh node from this pool would fit (only resource rejections are pool-specific).
	// If the rejection is taint/selector/affinity, a new node would have the same problem.
	// If the rejection is resources, a new empty node might have enough capacity.
	if dominantCategory == types.CategoryResources {
		// Check if a fresh node (max allocatable, zero requested) would fit
		freshNode := freshNodeFromPool(pool)
		freshFit := checkNodeFit(pod, freshNode)

		asStatus := lookupAutoscalerStatus(pool.Name, autoscalerStatus)
		diag.AutoscalerStatus = asStatus

		if !freshFit.rejected {
			// A new node would fit. Check autoscaler state.
			if asStatus != nil {
				if asStatus.IsNoStock() {
					diag.Verdict = types.VerdictNoStock
					diag.Reason = fmt.Sprintf("autoscaler backoff - %s", asStatus.BackoffErrorMessage)
					return diag
				}
				if asStatus.IsAtMaxSize() {
					diag.Verdict = types.VerdictRejected
					diag.Reason = fmt.Sprintf("all nodes full, autoscaler at max size (%d/%d)", asStatus.CloudProviderTarget, asStatus.MaxSize)
					return diag
				}
				if asStatus.IsBackoff() {
					diag.Verdict = types.VerdictCandidate
					diag.Reason = fmt.Sprintf("would fit on new node, but autoscaler in backoff (%s)", asStatus.BackoffErrorMessage)
					return diag
				}
			}
			// Autoscaler healthy or unknown
			diag.Verdict = types.VerdictCandidate
			diag.Reason = "would fit on new node, waiting for scale-up"
			return diag
		}
		// Even a fresh node can't fit
		diag.Verdict = types.VerdictRejected
		diag.Reason = fmt.Sprintf("%s (even on empty node)", freshFit.reason)
		return diag
	}

	// Taint/selector/affinity rejection: a new node would have the same issue
	diag.Verdict = types.VerdictRejected
	diag.Reason = dominantReason

	// Enrich with autoscaler status if available
	asStatus := lookupAutoscalerStatus(pool.Name, autoscalerStatus)
	if asStatus != nil {
		diag.AutoscalerStatus = asStatus
		if asStatus.IsNoStock() {
			diag.Verdict = types.VerdictNoStock
			diag.Reason = fmt.Sprintf("%s; autoscaler backoff - %s", dominantReason, asStatus.BackoffErrorMessage)
		}
	}

	return diag
}

type fitResult struct {
	rejected bool
	category types.RejectionCategory
	reason   string
}

// checkNodeFit checks all scheduling dimensions in priority order and returns the first rejection.
func checkNodeFit(pod *corev1.Pod, node types.NodeInfo) fitResult {
	// 1. Taints
	if reason, rejected := checkTaints(pod, node); rejected {
		return fitResult{rejected: true, category: types.CategoryTaint, reason: reason}
	}

	// 2. NodeSelector
	if reason, rejected := checkNodeSelector(pod, node); rejected {
		return fitResult{rejected: true, category: types.CategoryNodeSelector, reason: reason}
	}

	// 3. Node affinity
	if reason, rejected := checkNodeAffinity(pod, node); rejected {
		return fitResult{rejected: true, category: types.CategoryAffinity, reason: reason}
	}

	// 4. Resources
	if reason, rejected := checkResources(pod, node); rejected {
		return fitResult{rejected: true, category: types.CategoryResources, reason: reason}
	}

	return fitResult{rejected: false}
}

// dominantRejection finds the most common rejection category across all nodes.
// Returns the category and the reason from the first node with that category.
func dominantRejection(results []nodeResult) (types.RejectionCategory, string) {
	counts := map[types.RejectionCategory]int{}
	firstReason := map[types.RejectionCategory]string{}

	for _, r := range results {
		if !r.rejected {
			continue
		}
		counts[r.rejectionCategory]++
		if _, exists := firstReason[r.rejectionCategory]; !exists {
			firstReason[r.rejectionCategory] = r.reason
		}
	}

	var bestCategory types.RejectionCategory
	bestCount := 0
	for cat, count := range counts {
		if count > bestCount || (count == bestCount && cat < bestCategory) {
			bestCategory = cat
			bestCount = count
		}
	}

	return bestCategory, firstReason[bestCategory]
}

// freshNodeFromPool creates a synthetic node representing a freshly provisioned node
// from this nodepool. Uses the max allocatable resources from existing nodes and zero requests.
func freshNodeFromPool(pool types.NodepoolInfo) types.NodeInfo {
	fresh := types.NodeInfo{
		Name:   "<new-node>",
		Labels: map[string]string{},
	}

	if len(pool.Nodes) == 0 {
		return fresh
	}

	// Use the first node as a template for labels, taints, and allocatable resources.
	// All nodes in a nodepool should have the same allocatable (same instance type).
	template := pool.Nodes[0]
	fresh.Labels = template.Labels
	fresh.Taints = template.Taints
	fresh.AllocatableCPU = template.AllocatableCPU.DeepCopy()
	fresh.AllocatableMem = template.AllocatableMem.DeepCopy()
	fresh.AllocatableGPU = template.AllocatableGPU.DeepCopy()
	// Requested is zero (fresh node)

	return fresh
}

// lookupAutoscalerStatus finds the autoscaler status for a nodepool.
func lookupAutoscalerStatus(nodepoolName string, status map[string]*types.AutoscalerNodeGroupStatus) *types.AutoscalerNodeGroupStatus {
	if status == nil {
		return nil
	}
	return status[nodepoolName]
}

// formatQuantity formats a resource quantity for human display.
func formatQuantity(q resource.Quantity) string {
	return q.String()
}
```

```go
import (
	"fmt"

// rawAutoscalerStatus is the raw YAML structure of the cluster-autoscaler-status ConfigMap.
type rawAutoscalerStatus struct {
	Time              string `yaml:"time"`
	AutoscalerStatus  string `yaml:"autoscalerStatus"`
	ClusterWide       struct {
		Health struct {
			Status string `yaml:"status"`
		} `yaml:"health"`
		ScaleUp struct {
			Status string `yaml:"status"`
		} `yaml:"scaleUp"`
		ScaleDown struct {
			Status string `yaml:"status"`
		} `yaml:"scaleDown"`
	} `yaml:"clusterWide"`
	NodeGroups []rawNodeGroup `yaml:"nodeGroups"`
}

type rawNodeGroup struct {
	Name   string `yaml:"name"`
	Health struct {
		Status     string `yaml:"status"`
		NodeCounts struct {
			Registered struct {
				Total int `yaml:"total"`
				Ready int `yaml:"ready"`
			} `yaml:"registered"`
		} `yaml:"nodeCounts"`
		CloudProviderTarget int `yaml:"cloudProviderTarget"`
		MinSize             int `yaml:"minSize"`
		MaxSize             int `yaml:"maxSize"`
	} `yaml:"health"`
	ScaleUp struct {
		Status     string `yaml:"status"`
		BackoffInfo struct {
			ErrorCode    string `yaml:"errorCode"`
			ErrorMessage string `yaml:"errorMessage"`
		} `yaml:"backoffInfo"`
	} `yaml:"scaleUp"`
	ScaleDown struct {
		Status string `yaml:"status"`
	} `yaml:"scaleDown"`
}

// ParseAutoscalerStatus parses the YAML content of the cluster-autoscaler-status ConfigMap.
func ParseAutoscalerStatus(data string) (*types.AutoscalerStatus, error) {
	if data == "" {
		return nil, fmt.Errorf("empty autoscaler status data")
	}

	var raw rawAutoscalerStatus
	if err := yaml.Unmarshal([]byte(data), &raw); err != nil {
		return nil, fmt.Errorf("parsing autoscaler status YAML: %w", err)
	}

	status := &types.AutoscalerStatus{
		Time:             raw.Time,
		Status:           raw.AutoscalerStatus,
		ClusterHealth:    types.HealthStatus(raw.ClusterWide.Health.Status),
		ClusterScaleUp:   types.ScaleUpStatus(raw.ClusterWide.ScaleUp.Status),
		ClusterScaleDown: types.ScaleDownStatus(raw.ClusterWide.ScaleDown.Status),
	}

	for _, ng := range raw.NodeGroups {
		status.NodeGroups = append(status.NodeGroups, types.AutoscalerNodeGroupStatus{
			Name:                ng.Name,
			Health:              types.HealthStatus(ng.Health.Status),
			ReadyNodes:          ng.Health.NodeCounts.Registered.Ready,
			CloudProviderTarget: ng.Health.CloudProviderTarget,
			MinSize:             ng.Health.MinSize,
			MaxSize:             ng.Health.MaxSize,
			ScaleUp:             types.ScaleUpStatus(ng.ScaleUp.Status),
			ScaleDown:           types.ScaleDownStatus(ng.ScaleDown.Status),
			BackoffErrorCode:    ng.ScaleUp.BackoffInfo.ErrorCode,
			BackoffErrorMessage: ng.ScaleUp.BackoffInfo.ErrorMessage,
		})
	}

	return status, nil
}
```

```go
// BuildNodepoolScalingGroupMap builds a mapping from nodepool ID to ESS scaling group ID.
// ACK model: 1 nodepool = 1 ESS scaling group. The mapping is built by matching
// the number of nodes per nodepool to the readyNodes count per scaling group.
// This is a heuristic for v0.0.1. Returns an empty map if mapping cannot be determined.
func BuildNodepoolScalingGroupMap(nodes []types.NodeInfo, autoscalerStatus *types.AutoscalerStatus, nodepoolLabel string) map[string]string {
	if autoscalerStatus == nil || len(autoscalerStatus.NodeGroups) == 0 {
		return map[string]string{}
	}

	// Count nodes per nodepool
	nodepoolCounts := map[string]int{}
	for _, node := range nodes {
		if node.NodepoolID != "" {
			nodepoolCounts[node.NodepoolID]++
		}
	}

	// Build a map of scaling group -> ready node count
	sgCounts := map[string]int{}
	for _, ng := range autoscalerStatus.NodeGroups {
		sgCounts[ng.Name] = ng.ReadyNodes
	}

	// Match by node count. Track used scaling groups to avoid double-mapping.
	mapping := map[string]string{}
	usedSGs := map[string]bool{}

	for npID, npCount := range nodepoolCounts {
		for sgName, sgCount := range sgCounts {
			if usedSGs[sgName] {
				continue
			}
			if npCount == sgCount {
				mapping[npID] = sgName
				usedSGs[sgName] = true
				break
			}
		}
	}

	return mapping
}

// BuildAutoscalerStatusByNodepool maps nodepool names to their autoscaler status
// using the nodepool-to-scaling-group mapping.
func BuildAutoscalerStatusByNodepool(autoscalerStatus *types.AutoscalerStatus, npToSG map[string]string) map[string]*types.AutoscalerNodeGroupStatus {
	if autoscalerStatus == nil || npToSG == nil {
		return map[string]*types.AutoscalerNodeGroupStatus{}
	}

	// Index scaling groups by name
	sgIndex := map[string]*types.AutoscalerNodeGroupStatus{}
	for i := range autoscalerStatus.NodeGroups {
		sgIndex[autoscalerStatus.NodeGroups[i].Name] = &autoscalerStatus.NodeGroups[i]
	}

	result := map[string]*types.AutoscalerNodeGroupStatus{}
	for npID, sgName := range npToSG {
		if sg, ok := sgIndex[sgName]; ok {
			result[npID] = sg
		}
	}

	return result
}
```

```go
)

// PodReconcilerOpts holds configuration for the PodReconciler.
type PodReconcilerOpts struct {
	NodepoolLabel       string
	RecheckInterval     time.Duration
	InitialDelay        time.Duration
	AutoscalerConfigMap string
}

// PodReconciler reconciles Pending pods and emits per-nodepool diagnostic events.
type PodReconciler struct {
	client.Client
	Recorder        record.EventRecorder
	NodepoolLabel   string
	RecheckInterval time.Duration
	InitialDelay    time.Duration
	AutoscalerCM    string
}

// SetupPodReconciler creates and registers the PodReconciler with the manager.
func SetupPodReconciler(mgr ctrl.Manager, opts PodReconcilerOpts) error {
	r := &PodReconciler{
		Client:          mgr.GetClient(),
		Recorder:        mgr.GetEventRecorderFor("scheduler-debugger"),
		NodepoolLabel:   opts.NodepoolLabel,
		RecheckInterval: opts.RecheckInterval,
		InitialDelay:    opts.InitialDelay,
		AutoscalerCM:    opts.AutoscalerConfigMap,
	}

	return ctrl.NewControllerManagedBy(mgr).
		For(&corev1.Pod{}).
		WithEventFilter(predicate.Funcs{
			CreateFunc: func(e event.CreateEvent) bool {
				pod, ok := e.Object.(*corev1.Pod)
				if !ok {
					return false
				}
				return pod.Status.Phase == corev1.PodPending
			},
			UpdateFunc: func(e event.UpdateEvent) bool {
				pod, ok := e.ObjectNew.(*corev1.Pod)
				if !ok {
					return false
				}
				return pod.Status.Phase == corev1.PodPending
			},
			DeleteFunc:  func(_ event.DeleteEvent) bool { return false },
			GenericFunc: func(_ event.GenericEvent) bool { return false },
		}).
		WithOptions(controller.Options{
			MaxConcurrentReconciles: 1,
		}).
		Complete(r)
}

// Reconcile processes a single Pending pod.
func (r *PodReconciler) Reconcile(ctx context.Context, req reconcile.Request) (reconcile.Result, error) {
	logger := log.FromContext(ctx)

	var pod corev1.Pod
	if err := r.Get(ctx, req.NamespacedName, &pod); err != nil {
		return reconcile.Result{}, client.IgnoreNotFound(err)
	}

	// Only process Pending pods
	if pod.Status.Phase != corev1.PodPending {
		return reconcile.Result{}, nil
	}

	// Respect initial delay
	if r.InitialDelay > 0 {
		age := time.Since(pod.CreationTimestamp.Time)
		if age < r.InitialDelay {
			remaining := r.InitialDelay - age
			logger.V(1).Info("waiting for initial delay", "pod", req.NamespacedName, "remaining", remaining)
			return reconcile.Result{RequeueAfter: remaining}, nil
		}
	}

	// List all nodes
	var nodeList corev1.NodeList
	if err := r.List(ctx, &nodeList); err != nil {
		logger.Error(err, "failed to list nodes")
		return reconcile.Result{}, err
	}

	// Build node info list with current resource usage
	nodes, err := r.buildNodeInfos(ctx, nodeList.Items)
	if err != nil {
		logger.Error(err, "failed to build node infos")
		return reconcile.Result{}, err
	}

	// Group nodes by nodepool
	pools := diagnosis.GroupNodesByNodepool(nodes, r.NodepoolLabel)

	// Try to read autoscaler ConfigMap
	autoscalerStatusMap := r.getAutoscalerStatusByNodepool(ctx, nodes)

	// Run diagnosis
	results := diagnosis.DiagnoseNodepools(&pod, pools, autoscalerStatusMap)

	// Emit events
	for _, diag := range results {
		r.Recorder.Event(&pod, diag.EventType(), diag.EventReason(), diag.Message())
	}

	logger.V(1).Info("diagnosed pod", "pod", req.NamespacedName, "nodepools", len(results))

	return reconcile.Result{RequeueAfter: r.RecheckInterval}, nil
}

// buildNodeInfos converts K8s Node objects to NodeInfo structs.
func (r *PodReconciler) buildNodeInfos(ctx context.Context, nodes []corev1.Node) ([]dtypes.NodeInfo, error) {
	result := make([]dtypes.NodeInfo, 0, len(nodes))

	for _, node := range nodes {
		// Skip unschedulable nodes
		if node.Spec.Unschedulable {
			continue
		}

		// Skip nodes that are not Ready
		if !isNodeReady(&node) {
			continue
		}

		ni := dtypes.NodeInfo{
			Name:       node.Name,
			NodepoolID: node.Labels[r.NodepoolLabel],
			Taints:     node.Spec.Taints,
			Labels:     node.Labels,
		}

		// Set allocatable resources
		if cpu, ok := node.Status.Allocatable[corev1.ResourceCPU]; ok {
			ni.AllocatableCPU = cpu
		}
		if mem, ok := node.Status.Allocatable[corev1.ResourceMemory]; ok {
			ni.AllocatableMem = mem
		}
		if gpu, ok := node.Status.Allocatable[corev1.ResourceName("nvidia.com/gpu")]; ok {
			ni.AllocatableGPU = gpu
		}

		// Calculate requested resources by summing all non-terminated pod requests on this node
		requestedCPU, requestedMem, requestedGPU, err := r.getNodeRequestedResources(ctx, node.Name)
		if err != nil {
			return nil, fmt.Errorf("getting requested resources for node %s: %w", node.Name, err)
		}
		ni.RequestedCPU = requestedCPU
		ni.RequestedMem = requestedMem
		ni.RequestedGPU = requestedGPU

		result = append(result, ni)
	}

	return result, nil
}

// getNodeRequestedResources sums the resource requests of all non-terminal pods on a node.
func (r *PodReconciler) getNodeRequestedResources(ctx context.Context, nodeName string) (cpu, mem, gpu resource.Quantity, err error) {
	var podList corev1.PodList
	if err = r.List(ctx, &podList, client.MatchingFields{"spec.nodeName": nodeName}); err != nil {
		// Fallback: list all pods and filter. Field selectors may not work with fake client.
		var allPods corev1.PodList
		if err = r.List(ctx, &allPods); err != nil {
			return
		}
		for _, p := range allPods.Items {
			if p.Spec.NodeName != nodeName {
				continue
			}
			if p.Status.Phase == corev1.PodSucceeded || p.Status.Phase == corev1.PodFailed {
				continue
			}
			addPodRequests(&p, &cpu, &mem, &gpu)
		}
		err = nil
		return
	}

	for _, p := range podList.Items {
		if p.Status.Phase == corev1.PodSucceeded || p.Status.Phase == corev1.PodFailed {
			continue
		}
		addPodRequests(&p, &cpu, &mem, &gpu)
	}
	return
}

func addPodRequests(pod *corev1.Pod, cpu, mem, gpu *resource.Quantity) {
	for _, c := range pod.Spec.Containers {
		if req, ok := c.Resources.Requests[corev1.ResourceCPU]; ok {
			cpu.Add(req)
		}
		if req, ok := c.Resources.Requests[corev1.ResourceMemory]; ok {
			mem.Add(req)
		}
		if req, ok := c.Resources.Requests[corev1.ResourceName("nvidia.com/gpu")]; ok {
			gpu.Add(req)
		}
	}
}

// isNodeReady checks if a node has the Ready condition set to True.
func isNodeReady(node *corev1.Node) bool {
	for _, cond := range node.Status.Conditions {
		if cond.Type == corev1.NodeReady {
			return cond.Status == corev1.ConditionTrue
		}
	}
	return false
}

// getAutoscalerStatusByNodepool reads the autoscaler ConfigMap and builds a nodepool -> status map.
func (r *PodReconciler) getAutoscalerStatusByNodepool(ctx context.Context, nodes []dtypes.NodeInfo) map[string]*dtypes.AutoscalerNodeGroupStatus {
	logger := log.FromContext(ctx)

	var cm corev1.ConfigMap
	cmKey := client.ObjectKey{Namespace: "kube-system", Name: r.AutoscalerCM}
	if err := r.Get(ctx, cmKey, &cm); err != nil {
		logger.V(1).Info("autoscaler ConfigMap not found, skipping autoscaler correlation", "error", err)
		return nil
	}

	statusYAML, ok := cm.Data["status"]
	if !ok {
		logger.V(1).Info("autoscaler ConfigMap has no 'status' key")
		return nil
	}

	autoscalerStatus, err := diagnosis.ParseAutoscalerStatus(statusYAML)
	if err != nil {
		logger.Error(err, "failed to parse autoscaler status")
		return nil
	}

	npToSG := diagnosis.BuildNodepoolScalingGroupMap(nodes, autoscalerStatus, r.NodepoolLabel)
	return diagnosis.BuildAutoscalerStatusByNodepool(autoscalerStatus, npToSG)
}
```

```go
package main

import (
	"flag"
	"os"
	"time"

	corev1 "k8s.io/api/core/v1"
	"k8s.io/apimachinery/pkg/runtime"
	utilruntime "k8s.io/apimachinery/pkg/util/runtime"
	clientgoscheme "k8s.io/client-go/kubernetes/scheme"
	ctrl "sigs.k8s.io/controller-runtime"
	"sigs.k8s.io/controller-runtime/pkg/cache"
	"sigs.k8s.io/controller-runtime/pkg/client"
	"sigs.k8s.io/controller-runtime/pkg/log/zap"
	metricsserver "sigs.k8s.io/controller-runtime/pkg/metrics/server"

	"github.com/helmi/fitcheck/internal/controller"
)

var scheme = runtime.NewScheme()

func init() {
	utilruntime.Must(clientgoscheme.AddToScheme(scheme))
}

func main() {
	var (
		nodepoolLabel       string
		recheckInterval     string
		initialDelay        string
		namespace           string
		autoscalerConfigMap string
		metricsAddr         string
	)

	flag.StringVar(&nodepoolLabel, "nodepool-label", "alibabacloud.com/nodepool-id", "Label key for grouping nodes into nodepools")
	flag.StringVar(&recheckInterval, "recheck-interval", "30s", "Re-evaluation interval for pending pods")
	flag.StringVar(&initialDelay, "initial-delay", "10s", "Delay before first diagnosis")
	flag.StringVar(&namespace, "namespace", "", "Restrict to specific namespace (empty = all)")
	flag.StringVar(&autoscalerConfigMap, "autoscaler-configmap", "cluster-autoscaler-status", "ConfigMap name for autoscaler status")
	flag.StringVar(&metricsAddr, "metrics-addr", ":8080", "Metrics bind address")

	opts := zap.Options{Development: true}
	opts.BindFlags(flag.CommandLine)
	flag.Parse()

	ctrl.SetLogger(zap.New(zap.UseFlagOptions(&opts)))
	log := ctrl.Log.WithName("setup")

	recheckDuration, err := time.ParseDuration(recheckInterval)
	if err != nil {
		log.Error(err, "invalid --recheck-interval")
		os.Exit(1)
	}
	initialDelayDuration, err := time.ParseDuration(initialDelay)
	if err != nil {
		log.Error(err, "invalid --initial-delay")
		os.Exit(1)
	}

	cacheOpts := cache.Options{
		ByObject: map[client.Object]cache.ByObject{
			&corev1.ConfigMap{}: {
				Namespaces: map[string]cache.Config{
					"kube-system": {},
				},
			},
		},
	}

	mgr, err := ctrl.NewManager(ctrl.GetConfigOrDie(), ctrl.Options{
		Scheme:  scheme,
		Cache:   cacheOpts,
		Metrics: metricsserver.Options{BindAddress: metricsAddr},
	})
	if err != nil {
		log.Error(err, "unable to create manager")
		os.Exit(1)
	}

	if err := controller.SetupPodReconciler(mgr, controller.PodReconcilerOpts{
		NodepoolLabel:       nodepoolLabel,
		RecheckInterval:     recheckDuration,
		InitialDelay:        initialDelayDuration,
		AutoscalerConfigMap: autoscalerConfigMap,
	}); err != nil {
		log.Error(err, "unable to setup controller")
		os.Exit(1)
	}

	log.Info("starting manager")
	if err := mgr.Start(ctrl.SetupSignalHandler()); err != nil {
		log.Error(err, "manager exited with error")
		os.Exit(1)
	}
}
```
